<script>
var GameState =
{
	SPLASH: 1,
	TITLE: 2,
	OPTIONS: 3,
	CREDITS: 4,
	GAME: 5,
	LOSS: 6,
	RESET: 7,
	LVLCHANGE: 8,
	EXIT: 9,
	state: 0;
};

var statecounter: 0;

//Set up the canvas and drawing surface
var canvas = document.querySelector("canvas");
var drawingSurface = canvas.getContext("2d");

//Change the Game's State by pressing a Key
window.addEventListener("keydown", StateHandler, false);

function StateHandler(event)
{
	//When a key is pressed, change the Game's State
	GameState.state = GameState.state + 1;
	If (GameState.state > 9)
	{
		GameState.state = 1;
	}
	
	switch(GameState.state)
	{
		case 1:
			alert(My game state is in the SPLASH state);
		break;
		
		case 2:
			alert(My game state is in the TITLE state);
		break;
		
		case 3:
			alert(My game state is in the OPTIONS state);
		break;
		
		case 4:
			alert(My game state is in the CREDITS state);
		break;
		
		case 5:
			alert(My game state is in the GAME state);
		break;
		
		case 6:
			alert(My game state is in the LOSS state);
		break;
		
		case 7:
			alert(My game state is in the RESET state);
		break;
		
		case 8:
			alert(My game state is in the LVLCHANGE state);
		break;
		
		case 9:
			alert(My game state is in the EXIT state);
		break;
	}

</script>
